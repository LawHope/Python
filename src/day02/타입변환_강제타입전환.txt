	  종류들▼
	 %d : decimal 정수표기용
	 %f : Float 실수표기용
	 %c : Character 문자 하나 표기할때
	 %s : Stgring 문자 열을 표기할때

>>>>>fmt는 라이브러리가 아닌 외부 개발자가 만든 소스 (지금 쓰는 스캐너)라서 알려주는 기호임

		//변수선언
		String name; //문자이름이라 String

		char gra; //단 학점은 A,B,C,D,F
		



		//총점, 평균 처리

		avr = tot /3;
		

		//처리

		System.out.println("이름은? "); //입력 받고 넘어감

		name = sc.next();  //이름을 텍스트로 입력받는 거라 name 변수에 대입/ next();를 씀
		System.out.println("국어는? ");
		kor = sc.nextInt(); //실행자가 입력받은 값을 변수에 대입/ nextInt();숫자라서Int

		mat = sc.nextInt();
		System.out.println("영어는? ");
		eng = sc.nextInt();
